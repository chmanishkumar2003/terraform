pipeline{
 
    agent any
 
    stages{
 
        stage("clone from github"){
 
            steps{
 
               git 'https://github.com/akshu20791/addressbook-cicd-project/'
 
               echo 'clone done'
 
            }
 
        }
 
        stage("compile the project"){
 
            steps{
 
                sh 'mvn compile'
 
            }
 
        }
 
        stage("run the test cases"){
 
         steps{
 
            sh 'mvn test'
 
         }
 
        }
 
        stage("package the project"){
 
            steps{
 
                sh 'mvn package'
 
            }
 
        }
 
        stage("deploying the project to tomcat"){
 
            steps{
  sh "sudo cp /var/lib/jenkins/workspace/myfirstprojectdeployment/target/addressbook.war /home/ubuntu/apache-tomcat-9.0.118/webapps"
 
            }
 
        }
 
    }
 
}
